357 #if SDO_ACCESS
358 void read_sdo(ec_sdo_request_t *req)
359 {
360  switch (ecrt_sdo_request_state(req)) {
361  case EC_REQUEST_UNUSED: // request was not used yet
362  ecrt_sdo_request_read(req); // trigger first read
363  break;
364  case EC_REQUEST_BUSY:
365  fprintf(stderr, "SDO still busy...\n");
366  break;
367  case EC_REQUEST_SUCCESS:
368  logmsg(1, "SDO value read: 0x%X\n",
369  EC_READ_U32(ecrt_sdo_request_data(req)));
370  ecrt_sdo_request_read(req); // trigger next read
371  break;
372  case EC_REQUEST_ERROR:
373  fprintf(stderr, "Failed to read SDO!\n");
374  ecrt_sdo_request_read(req); // retry reading
375  break;
376  }
377 }

388  fprintf(stderr, "SDO write still busy...\n");
389  break;
390  case EC_REQUEST_SUCCESS:
391  logmsg(1, "SDO value written: 0x%X\n", data);
392  ecrt_sdo_request_write(req); // trigger next read ???
393  break;
394  case EC_REQUEST_ERROR:

396  ecrt_sdo_request_write(req); // retry writing

404 void cyclic_task()
405 {
406  /* sync the dc clock of the slaves */
407  ecrt_master_sync_slave_clocks(master);
408 
409  // receive process data
410  ecrt_master_receive(master);
411  ecrt_domain_process(domain1);
412 
413  // check process data state (optional)
414  check_domain1_state();
415 
416  if (counter) {
417  counter--;
418  } else { // do this at 1 Hz
419  counter = FREQUENCY;
420 
421  // calculate new process data
422  blink = !blink;
423 
424  // check for master state (optional)
425  check_master_state();
426 
427  // check for islave configuration state(s) (optional)
428  check_slave_config_states();
429 
430 #if SDO_ACCESS
431  // read process data SDO
432  read_sdo(sdo);
433  read_sdo(request[0]);
434  read_sdo(request[1]);
435  read_sdo(request[2]);
436 
437  write_sdo(sdo_download_requests[0], sdoexample); /* SDO download value to the node */
438 #endif
439  }
440 
441  /* Read process data */
442  unsigned int sn_status = EC_READ_U16(domain1_pd + off_pdo1_in);
443  unsigned int sn_modes = EC_READ_U8(domain1_pd + off_pdo2_in);
444  unsigned int sn_position = EC_READ_U32(domain1_pd + off_pdo3_in);
445  unsigned int sn_velocity = EC_READ_U32(domain1_pd + off_pdo4_in);
446  unsigned int sn_torque = EC_READ_U16(domain1_pd + off_pdo5_in);
447 
448  logmsg(2, "[REC] 0x%4x 0x%4x 0x%8x 0x%8x 0x%4x\n",
449  sn_status, sn_modes,
450  sn_position, sn_velocity, sn_torque);
451 
452 #if 0
453  // read process data
454  printf("AnaIn: state %u value %u\n",
455  EC_READ_U8(domain1_pd + off_ana_in_status),
456  EC_READ_U16(domain1_pd + off_ana_in_value));
457 #endif
458 
459  // write process data
460  //EC_WRITE_U8(domain1_pd + off_dig_out, blink ? 0x06 : 0x09);
461 
462 #ifdef CIA402
463 #define STATUSW1 0x88AA
464 #define STATUSW2 0xAA88
465 #define OPMODES1 0xf1
466 #define OPMODES2 0x1f
467 #define TORVAL1 0xabab
468 #define TORVAL2 0xbaba
469 #define VELVAL1 0x2d2d4d4d
470 #define VELVAL2 0xd4d4d2d2
471 #define POSVAL1 0xe4e4e2e2
472 #define POSVAL2 0x2e2e4e4e
473 
474  EC_WRITE_U16(domain1_pd + off_pdo1_out, (blink ? STATUSW1 : STATUSW2)&0xffff);
475  EC_WRITE_U8(domain1_pd + off_pdo2_out, (blink ? OPMODES1 : OPMODES2)&0xff);
476  EC_WRITE_U16(domain1_pd + off_pdo3_out, (blink ? TORVAL1 : TORVAL2)&0xffff);
477  EC_WRITE_U32(domain1_pd + off_pdo4_out, blink ? POSVAL1 : POSVAL2);
478  EC_WRITE_U32(domain1_pd + off_pdo5_out, blink ? VELVAL1 : VELVAL2);
479 #else

488 
489  // send process data
490  ecrt_domain_queue(domain1);

492  //printf("Wrote %x to slave\n", blink ? TESTWORD1 : TESTWORD2);
493 }

544 static void cmdline(int argc, char **argv)
545 {
546  int flags, opt;
547  int nsecs, tfnd;
548 
549  const char *options = "hvl:";
550 
551  nsecs = 0;
552  tfnd = 0;
553  flags = 0;
554 
555  while ((opt = getopt(argc, argv, options)) != -1) {
556  switch (opt) {
557  case 'v':
558  printversion(argv[0]);
559  exit(0);
560  break;
561 
562  case 'l':
563  g_dbglvl = atoi(optarg);
564  if (g_dbglvl<0 || g_dbglvl>MAXDBGLVL) {
565  fprintf(stderr, "Error unsuported debug level %d.\n", g_dbglvl);
566  exit(1);
567  }
568  break;
569 
570  case 'h':
571  default:
572  printhelp(argv[0]);
573  exit(1);
574  break;
575  }
576  }
