105 // Send some data back for a HTTP request
106 void httpd_send(chanend tcp_svr, xtcp_connection_t *conn)
107 {
108  struct httpd_state_t *hs = (struct httpd_state_t *) conn->appstate;
109 
110  // Check if we need to resend previous data
111  if (conn->event == XTCP_RESEND_DATA) {
112  xtcp_send(tcp_svr, hs->prev_dptr, (hs->dptr - hs->prev_dptr));
113  return;
114  }
115 
116  // Check if we have no data to send
117  if (hs->dlen == 0 || hs->dptr == NULL)
118  {
119  // Terminates the send process
120  xtcp_complete_send(tcp_svr);
121  // Close the connection
122  xtcp_close(tcp_svr, conn);
123  }
124  // We need to send some new data
125  else {
126  int len = hs->dlen;
127 
128  if (len > conn->mss)
129  len = conn->mss;
130 
131  xtcp_send(tcp_svr, hs->dptr, len);
132 
133  hs->prev_dptr = hs->dptr;
134  hs->dptr += len;
135  hs->dlen -= len;
136  }

142 // Setup a new connection
143 void httpd_init_state(chanend tcp_svr, xtcp_connection_t *conn)
144 {
145  int i;
146 
147  // Try and find an empty connection slot
148  for (i=0;i<NUM_HTTPD_CONNECTIONS;i++)
149  {
150  if (!connection_states[i].active)
151  break;
152  }
153 
154  // If no free connection slots were found, abort the connection
155  if ( i == NUM_HTTPD_CONNECTIONS )
156  {
157  xtcp_abort(tcp_svr, conn);
158  }
159  // Otherwise, assign the connection to a slot //
160  else
161  {
162  connection_states[i].active = 1;
163  connection_states[i].conn_id = conn->id;
164  connection_states[i].dptr = NULL;
165  xtcp_set_connection_appstate(
166  tcp_svr,
167  conn,
168  (xtcp_appstate_t) &connection_states[i]);
169  }

189 // HTTP event handler
190 void httpd_handle_event(chanend tcp_svr, xtcp_connection_t *conn)
191 {
192  // We have received an event from the TCP stack, so respond
193  // appropriately
194 
195  // Ignore events that are not directly relevant to http
196  switch (conn->event)
197  {
198  case XTCP_IFUP:
199  case XTCP_IFDOWN:
200  case XTCP_ALREADY_HANDLED:
201  return;
202  default:
203  break;
204  }
205 
206  // Check if the connection is a http connection
207  if (conn->local_port == 80) {
208  switch (conn->event)
209  {
210  case XTCP_NEW_CONNECTION:
211  httpd_init_state(tcp_svr, conn);
212  break;
213  case XTCP_RECV_DATA:
214  httpd_recv(tcp_svr, conn);
215  break;
216  case XTCP_SENT_DATA:
217  case XTCP_REQUEST_DATA:
218  case XTCP_RESEND_DATA:
219  httpd_send(tcp_svr, conn);
220  break;
221  case XTCP_TIMED_OUT:
222  case XTCP_ABORTED:
223  case XTCP_CLOSED:
224  httpd_free_state(conn);
225  break;
226  default:
227  // Ignore anything else
228  break;
229  }
230  conn->event = XTCP_ALREADY_HANDLED;
231  }
233  return;
234 }
