219 int build_icmp_response(unsigned char rxbuf[], unsigned char txbuf[], const unsigned char own_mac_addr[6])
220 {
221  static const unsigned char own_ip_addr[4] = OWN_IP_ADDRESS;
222  unsigned icmp_checksum;
223  int datalen;
224  int totallen;
225  const int ttl = 0x40;
226  int pad;
227 
228  // Precomputed empty IP header checksum (inverted, bytereversed and shifted right)
229  unsigned ip_checksum = 0x0185;
230 
231  for (int i = 0; i < 6; i++)
232  {
233  txbuf[i] = rxbuf[6 + i];
234  }
235  for (int i = 0; i < 4; i++)
236  {
237  txbuf[30 + i] = rxbuf[26 + i];
238  }
239 
240  // As far as I understood, the icmp_checksum has nothing to do with the received checksum (jes)
241  //icmp_checksum = byterev((rxbuf, const unsigned[])[9]) >> 16;
242  icmp_checksum = 0x0; /* start with all 0 */
243  for (int i = 0; i < 4; i++)
244  {
245  txbuf[38 + i] = rxbuf[38 + i];
246  }
247 
248  // FIXME datalen is too large by 2 bytes
249  totallen = byterev((rxbuf, const unsigned[])[4]) >> 16;
250  datalen = totallen - 28;
251  for (int i = 0; i < datalen; i++)
252  {
253  txbuf[42 + i] = rxbuf[42+i];
254  }
255 
256  for (int i = 0; i < 6; i++)
257  {
258  txbuf[6 + i] = own_mac_addr[i];
259  }
260  (txbuf, unsigned[])[3] = 0x00450008;
261  totallen = byterev(28 + datalen) >> 16;
262  (txbuf, unsigned[])[4] = totallen;
263  ip_checksum += totallen;
264  (txbuf, unsigned[])[5] = 0x01000000 | (ttl << 16);
265  (txbuf, unsigned[])[6] = 0;
266  for (int i = 0; i < 4; i++)
267  {
268  txbuf[26 + i] = own_ip_addr[i];
269  }
270  ip_checksum += (own_ip_addr[0] | own_ip_addr[1] << 8);
271  ip_checksum += (own_ip_addr[2] | own_ip_addr[3] << 8);
272  ip_checksum += txbuf[30] | (txbuf[31] << 8);
273  ip_checksum += txbuf[32] | (txbuf[33] << 8);
274 
275  txbuf[34] = 0x00;
276  txbuf[35] = 0x00;
277 
278 // jes: calculate the checksum:
279  // jes: first set the checksum to 0 in the header:
280  icmp_checksum = 0;
281  txbuf[36] = icmp_checksum >> 8;
282  txbuf[37] = icmp_checksum & 0xFF;
283 
284  // sum up the ICMP header
285  icmp_checksum += (unsigned)txbuf[34] | ((unsigned)txbuf[35]<<8); // should make 16-bit calculation
286  icmp_checksum += (unsigned)txbuf[36] | ((unsigned)txbuf[37]<<8); // should make 16-bit calculation
287  icmp_checksum += (unsigned)txbuf[38] | ((unsigned)txbuf[39]<<8); // should make 16-bit calculation
288  icmp_checksum += (unsigned)txbuf[40] | ((unsigned)txbuf[41]<<8); // should make 16-bit calculation
289 
290 // FIXME add checkk for odd number of bytes
291  // sum up the package content
292  for (int j=0; j<(datalen-2); j+=2) {
293  icmp_checksum += (unsigned)txbuf[42+j] | ((unsigned)txbuf[42+j+1]<<8);
294  }
295 
296  icmp_checksum = (icmp_checksum&0xffff) + (icmp_checksum>>16); // add carry bits
297  icmp_checksum = ~icmp_checksum; // one's complement
298 
299  // set the freshly calculated checksum in the header (byte swap):
300  txbuf[37] = icmp_checksum >> 8;
301  txbuf[36] = icmp_checksum & 0xFF;
302 
303  while (ip_checksum >> 16)
304  {
305  ip_checksum = (ip_checksum & 0xFFFF) + (ip_checksum >> 16);
306  }
307  ip_checksum = byterev(~ip_checksum) >> 16;
308  txbuf[24] = ip_checksum >> 8;
309  txbuf[25] = ip_checksum & 0xFF;
310 
311  for (pad = 42 + datalen; pad < 64; pad++)
312  {
313  txbuf[pad] = 0x00;
314  }
315  return pad;
316 }

319 int is_valid_icmp_packet(const unsigned char rxbuf[], int nbytes)
320 {
321  static const unsigned char own_ip_addr[4] = OWN_IP_ADDRESS;
322  unsigned totallen;
323 
324  if (rxbuf[12] != 0x08 || rxbuf[13] != 0x00) /* ICMP is a IP packet, so first check this */
325  return 0;
326 
327  if (rxbuf[23] != 0x01)
328  return 0;
329 
330  printstr("ICMP packet received\n");
331 
332  if ((rxbuf, const unsigned[])[3] != 0x00450008)
333  {
334  printstr("Invalid et_ver_hdrl_tos\n");
335  return 0;
336  }
337  if (((rxbuf, const unsigned[])[8] >> 16) != 0x0008)
338  {
339  printstr("Invalid type_code\n");
340  return 0;
341  }
342  for (int i = 0; i < 4; i++)
343  {
344  if (rxbuf[30 + i] != own_ip_addr[i])
345  {
346  printstr("Not for us\n");
347  return 0;
348  }
349  }
350 
351  totallen = byterev((rxbuf, const unsigned[])[4]) >> 16;
352  if (nbytes > 60 && nbytes != totallen + 12/*14*/)
353  {
354  printstr("Invalid size\n");
355  printintln(nbytes);
356  printintln(totallen+12/*14*/);
357  return 0;
358  }
359  if (checksum_ip(rxbuf) != 0)
360  {
361  printstr("Bad checksum\n");
362  return 0;
363  }
364 
365  return 1;
